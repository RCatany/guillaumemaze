% m2cols Transform a 1 dimensional matrix to a square one with elements on columns
%
% C = m2cols(C)
%
% Transform the matrix C(1,N) or C(N,1) to a square matrix
% of dimensions NxN with elements from C on each of the columns.
%
% Example:
% C = [1 ; 2 ; 4 ; 5]
% m2diag(C)
